Implementation sketch 
// components/ui/responsive-dialog.tsx
'use client'
import { useIsMobile } from '@/lib/hooks/use-is-mobile'
import * as Dialog from '@/components/ui/dialog'
import * as Sheet from '@/components/ui/sheet'

export function ResponsiveDialog({ open, onOpenChange, children }) {
  const isMobile = useIsMobile()
  if (isMobile) {
    return (
      <Sheet.Sheet open={open} onOpenChange={onOpenChange}>
        <Sheet.SheetContent side="bottom" className="rounded-t-3xl max-h-[85vh]">
          {children}
        </Sheet.SheetContent>
      </Sheet.Sheet>
    )
  }
  return (
    <Dialog.Dialog open={open} onOpenChange={onOpenChange}>
      <Dialog.DialogContent>{children}</Dialog.DialogContent>
    </Dialog.Dialog>
  )
}

A4 — iOS keyboard

Keep sticky bars above the keyboard

Every sticky-bottom action bar (modals + edit pages) covers up under the iOS keyboard by default. One viewport meta change fixes most of it; 100dvh instead of 100vh handles the rest. 

// In app/layout.tsx — viewport meta
export const viewport: Viewport = {
  width: 'device-width',
  initialScale: 1,
  viewportFit: 'cover',
  // NEW: tells iOS to resize the layout when the keyboard opens,
  // so sticky-bottom bars stay above the keyboard
  interactiveWidget: 'resizes-content',
}

// Use 100dvh (dynamic viewport) consistently:

Story 17.x: Mobile photo evidence capture

As a worksite user, I want to capture and attach photos as
evidence directly from my phone, so that I can document
compliance during site walks without leaving the building or
returning to a desktop.

AC 1: "Lägg till bevis" Sheet opens 4 options: Ta foto / Bibliotek
      / Fil / Länka. Mobile-only; desktop keeps current 3 buttons.
AC 2: "Ta foto" opens native back camera via capture="environment".
      Multi-shot supported.
AC 3: Photos compressed client-side (max 1920px, JPEG q=0.85).
      EXIF GPS stripped before upload.
AC 4: Each photo gets caption field + auto-association with the
      kravpunkt or list-item context.
AC 5: Photos render as thumbnails (Supabase transform) in
      LinkedArtifactsPanel.
AC 6: Storage usage delta shown in toast post-upload.
AC 7: Tests for compression utility, EXIF stripping, multi-shot
      stack management.

Out of scope (defer to Phase E): offline queue, AI photo
description from kravpunkt context, share-sheet integration.
