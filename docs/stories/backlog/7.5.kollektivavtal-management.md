## Story

**As an** HR manager,
**I want** to upload and assign kollektivavtal to employees,
**so that** the AI knows which agreement applies to each employee.

1. Kollektivavtal page created at `/hr/kollektivavtal`
2. Upload PDF button
3. Upload flow:
   - Select PDF (max 20MB)
   - Name input (e.g., "Byggnads Kollektivavtal 2024")

**Upload API:**

```typescript
// app/api/kollektivavtal/upload/route.ts
import { NextResponse } from 'next/server'
import { withWorkspace } from '@/lib/auth/with-workspace'
import { checkPermission, Permission } from '@/lib/permissions/roles'
import { WorkspaceRole } from '@prisma/client'
import { prisma } from '@/lib/prisma'
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
)

export async function POST(request: Request) {
  return withWorkspace(async ({ workspaceId, userId, role }) => {
    checkPermission(role as WorkspaceRole, Permission.MANAGE_EMPLOYEES)

    const formData = await request.formData()
    const file = formData.get('file') as File
    const name = formData.get('name') as string
    const type = formData.get('type') as string

    if (!file || !name || !type) {
      return NextResponse.json(
        { error: 'Missing required fields' },
        { status: 400 }
      )
    }

    // Upload PDF to Supabase Storage
    const fileName = `${workspaceId}/kollektivavtal/${Date.now()}-${file.name}`
    const { data: uploadData, error: uploadError } = await supabase.storage
      .from('kollektivavtal-pdfs')
      .upload(fileName, file, {
        cacheControl: '3600',
        upsert: false,
      })

    if (uploadError) {
      console.error('Supabase upload error:', uploadError)
      return NextResponse.json(
        { error: 'Failed to upload PDF' },
        { status: 500 }
      )
    }

    // Get public URL
    const { data: urlData } = supabase.storage
      .from('kollektivavtal-pdfs')
      .getPublicUrl(fileName)

    // Create kollektivavtal record
    const kollektivavtal = await prisma.kollektivavtal.create({
      data: {
        workspaceId,
        name,
        type,
        pdfUrl: urlData.publicUrl,
        pdfSize: file.size,
        uploadedBy: userId,
      },
    })

    return NextResponse.json({ kollektivavtalId: kollektivavtal.id })
  })
}
```

```typescript
// app/api/kollektivavtal/process/route.ts
import { NextResponse } from 'next/server'
import { withWorkspace } from '@/lib/auth/with-workspace'
import { prisma } from '@/lib/prisma'
import { embedKollektivavtal } from '@/lib/kollektivavtal/embed'

export async function POST(request: Request) {
  return withWorkspace(async ({ workspaceId }) => {
    const { kollektivavtalId } = await request.json()

    // Verify kollektivavtal belongs to workspace
    const kollektivavtal = await prisma.kollektivavtal.findFirst({
      where: { id: kollektivavtalId, workspaceId },
    })

    if (!kollektivavtal) {
      return NextResponse.json(
        { error: 'Kollektivavtal not found' },
        { status: 404 }
      )
    }

    // Process PDF: download, extract text, chunk, embed
    await embedKollektivavtal(kollektivavtal)

    return NextResponse.json({ success: true })
  })
}
```

```typescript
// lib/kollektivavtal/embed.ts
import { prisma } from '@/lib/prisma'
import { extractTextFromPDF } from '@/lib/pdf/extract-text'
import { chunkText } from '@/lib/embeddings/chunk-text'
import { generateEmbedding } from '@/lib/embeddings/generate-embedding'

export async function embedKollektivavtal(kollektivavtal: any) {
  // Download PDF
  const response = await fetch(kollektivavtal.pdfUrl)
  const arrayBuffer = await response.arrayBuffer()
  const buffer = Buffer.from(arrayBuffer)

  // Extract text from PDF
  const { text, pages } = await extractTextFromPDF(buffer)

  // Chunk text
  const chunks = chunkText(text, {
    maxChunkSize: 1000,
    overlap: 200,
  })

  // Generate embeddings and store
  for (let i = 0; i < chunks.length; i++) {
    const chunk = chunks[i]
    const embedding = await generateEmbedding(chunk.text)

    await prisma.$executeRaw`
      INSERT INTO kollektivavtal_chunks (id, kollektivavtal_id, content, embedding, chunk_index, page_number)
      VALUES (
        gen_random_uuid(),
        ${kollektivavtal.id}::uuid,
        ${chunk.text},
        ${embedding}::vector,
        ${i},
        ${chunk.pageNumber}
      )
    `
  }

  console.log(
    `Embedded ${chunks.length} chunks for kollektivavtal ${kollektivavtal.id}`
  )
}
```

```typescript
// lib/chat/retrieve-context.ts (updated)
export async function retrieveContext(
  query: string,
  workspaceId: string
): Promise<ContextChunk[]> {
  const queryEmbedding = await generateEmbedding(query)

  // Search both laws and kollektivavtal
  const [lawResults, kollektivavtalResults] = await Promise.all([
    // Search laws
    prisma.$queryRaw<ContextChunk[]>`
      SELECT
        lc.id,
        lc.content,
        lc.embedding <=> ${queryEmbedding}::vector as distance,
        l.sfs_nummer as source,
        l.rubrik as title,
        'law' as type
      FROM law_chunks lc
      JOIN workspace_laws wl ON lc.law_id = wl.law_id
      JOIN laws l ON wl.law_id = l.id
      WHERE wl.workspace_id = ${workspaceId}::uuid
        AND lc.embedding <=> ${queryEmbedding}::vector < 0.7
      ORDER BY distance ASC
      LIMIT 5
    `,

    // Search kollektivavtal
    prisma.$queryRaw<ContextChunk[]>`
      SELECT
        kc.id,
        kc.content,
        kc.embedding <=> ${queryEmbedding}::vector as distance,
        k.name as source,
        k.name as title,
        'kollektivavtal' as type
      FROM kollektivavtal_chunks kc
      JOIN kollektivavtal k ON kc.kollektivavtal_id = k.id
      WHERE k.workspace_id = ${workspaceId}::uuid
        AND kc.embedding <=> ${queryEmbedding}::vector < 0.7
      ORDER BY distance ASC
      LIMIT 3
    `,
  ])

  // Combine and sort by distance
  const allResults = [...lawResults, ...kollektivavtalResults].sort(
    (a, b) => (a.distance || 0) - (b.distance || 0)
  )

  return allResults.slice(0, 5)
}
```
