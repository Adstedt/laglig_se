# Story 7.2: Implement Employee Profile Page with Tabs

## Status

Draft

## Story

**As an** HR manager,
**I want** to view detailed employee profile,
**so that** I see all HR data and compliance status in one place.

## Acceptance Criteria

1. Clicking employee name opens profile page: `/hr/employees/[id]`
2. Profile tabs: Overview, Documents, Compliance, Activity
3. **Overview tab:**
   - Personal info (name, personnummer, email, phone)
   - Employment details (role, department, manager, employment date)
   - Contract type, end date (if fixed-term)
   - Edit button (opens modal with all fields)
4. **Documents tab:**
   - Uploaded documents (contract, ID, certificates)
   - Upload button (PDF/image)
   - Document list with: filename, upload date, uploader
   - Download/delete buttons
5. **Compliance tab:**
   - Kollektivavtal assignment
   - Compliance status (Compliant/Needs Attention/Non-Compliant)
   - Related laws (laws that apply to this employee)
6. **Activity tab:**
   - Audit log: Who edited this employee's data and when
7. Role-based access: Only HR Manager, Admin, Owner can access
8. Back button returns to Employee List
9. Mobile-responsive layout (tabs stack vertically on mobile)

```typescript
// lib/hr/get-employee.ts
import { prisma } from '@/lib/prisma'
import { withWorkspace } from '@/lib/auth/with-workspace'

export async function getEmployee(employeeId: string) {
  return withWorkspace(async ({ workspaceId }) => {
    const employee = await prisma.employee.findFirst({
      where: {
        id: employeeId,
        workspaceId,
      },
      include: {
        manager: {
          select: {
            id: true,
            name: true,
          },
        },
        subordinates: {
          select: {
            id: true,
            name: true,
            role: true,
          },
        },
        documents: {
          include: {
            uploader: {
              select: {
                name: true,
                email: true,
              },
            },
          },
          orderBy: {
            createdAt: 'desc',
          },
        },
        kollektivavtal: {
          include: {
            kollektivavtal: {
              select: {
                id: true,
                name: true,
                type: true,
              },
            },
          },
        },
        activities: {
          include: {
            user: {
              select: {
                name: true,
                email: true,
              },
            },
          },
          orderBy: {
            createdAt: 'desc',
          },
          take: 50, // Last 50 activities
        },
      },
    })

    if (!employee) {
      return null
    }

    // Decrypt personnummer for display
    const { decrypt } = await import('@/lib/encryption')
    const decryptedPersonnummer = await decrypt(employee.personnummer)

    return {
      ...employee,
      personnummer: decryptedPersonnummer,
    }
  })
}
```

```typescript
'use client'

import { useState } from 'react'
import { useRouter } from 'next/navigation'

interface DocumentsTabProps {
  employee: any
}

export function DocumentsTab({ employee }: DocumentsTabProps) {
  const router = useRouter()
  const [isUploading, setIsUploading] = useState(false)

  async function handleUpload(e: React.ChangeEvent<HTMLInputElement>) {
    const file = e.target.files?.[0]
    if (!file) return

    // Validate file type
    const allowedTypes = ['application/pdf', 'image/jpeg', 'image/png']
    if (!allowedTypes.includes(file.type)) {
      alert('Only PDF and images (JPG, PNG) are allowed')
      return
    }

    // Validate file size (max 10MB)
    if (file.size > 10 * 1024 * 1024) {
      alert('File size must be less than 10MB')
      return
    }

    setIsUploading(true)
    try {
      // Upload to Supabase Storage
      const formData = new FormData()
      formData.append('file', file)
      formData.append('employeeId', employee.id)

      const response = await fetch('/api/employees/documents', {
        method: 'POST',
        body: formData
      })

      if (response.ok) {
        router.refresh()
      } else {
        alert('Failed to upload document')
      }
    } catch (error) {
      alert('Failed to upload document')
    } finally {
      setIsUploading(false)
    }
  }

  async function handleDelete(documentId: string) {
    if (!confirm('Are you sure you want to delete this document?')) {
      return
    }

    try {
      const response = await fetch(`/api/employees/documents/${documentId}`, {
        method: 'DELETE'
      })

      if (response.ok) {
        router.refresh()
      } else {
        alert('Failed to delete document')
      }
    } catch (error) {
      alert('Failed to delete document')
    }
  }

  return (
    <div className="p-6">
      {/* Header with Upload Button */}
      <div className="flex items-center justify-between mb-6">
        <div>
          <h2 className="text-lg font-semibold">Documents</h2>

          {isUploading ? 'Uploading...' : 'Upload Document'}

            className="hidden"
            accept=".pdf,.jpg,.jpeg,.png"
            onChange={handleUpload}
            disabled={isUploading}
          />
        </label>
      </div>

        <div className="space-y-3">

            <div
              key={doc.id}
              className="flex items-center justify-between p-4 bg-gray-50 rounded-lg hover:bg-gray-100"
            >
              <div className="flex items-center gap-4">
                {/* File Icon */}
                <div className="w-10 h-10 bg-blue-100 rounded-lg flex items-center justify-center">
                  {doc.fileType === 'application/pdf' ? (
```

```typescript
function calculateComplianceStatus(employee: any) {
  const reasons: string[] = []

  // Check required fields
  if (!employee.personnummer) {
    reasons.push('Missing personnummer')
  }
  if (!employee.employmentDate) {
    reasons.push('Missing employment date')
  }

  // Check kollektivavtal assignment
  if (!employee.kollektivavtal || employee.kollektivavtal.length === 0) {
    reasons.push('No kollektivavtal assigned')
  }

  // Check documents (contract required)
  const hasContract = employee.documents?.some((doc: any) =>
    doc.fileName.toLowerCase().includes('contract') || doc.fileName.toLowerCase().includes('avtal')
  )
  if (!hasContract) {
    reasons.push('No contract document uploaded')
  }

  // Determine status
  if (reasons.length === 0) {
    return { status: 'compliant' as const, reasons: ['All requirements met'] }
  } else if (reasons.some((r) => r.includes('Missing personnummer') || r.includes('employment date'))) {
    return { status: 'non_compliant' as const, reasons }
  } else {
    return { status: 'needs_attention' as const, reasons }
  }
}
```

```typescript
// app/api/employees/documents/route.ts
import { NextResponse } from 'next/server'
import { withWorkspace } from '@/lib/auth/with-workspace'
import { checkPermission, Permission } from '@/lib/permissions/roles'
import { WorkspaceRole } from '@prisma/client'
import { prisma } from '@/lib/prisma'
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
)

export async function POST(request: Request) {
  return withWorkspace(async ({ workspaceId, userId, role }) => {
    checkPermission(role as WorkspaceRole, Permission.MANAGE_EMPLOYEES)

    const formData = await request.formData()
    const file = formData.get('file') as File
    const employeeId = formData.get('employeeId') as string

    if (!file || !employeeId) {
      return NextResponse.json(
        { error: 'Missing required fields' },
        { status: 400 }
      )
    }

    // Verify employee belongs to workspace
    const employee = await prisma.employee.findFirst({
      where: { id: employeeId, workspaceId },
    })

    if (!employee) {
      return NextResponse.json({ error: 'Employee not found' }, { status: 404 })
    }

    // Upload to Supabase Storage
    const fileName = `${workspaceId}/${employeeId}/${Date.now()}-${file.name}`
    const { data: uploadData, error: uploadError } = await supabase.storage
      .from('employee-documents')
      .upload(fileName, file, {
        cacheControl: '3600',
        upsert: false,
      })

    if (uploadError) {
      console.error('Supabase upload error:', uploadError)
      return NextResponse.json(
        { error: 'Failed to upload file' },
        { status: 500 }
      )
    }

    // Get public URL
    const { data: urlData } = supabase.storage
      .from('employee-documents')
      .getPublicUrl(fileName)

    // Save document record
    const document = await prisma.employeeDocument.create({
      data: {
        employeeId,
        fileName: file.name,
        fileSize: file.size,
        fileType: file.type,
        fileUrl: urlData.publicUrl,
        uploadedBy: userId,
      },
    })

    // Log activity
    await prisma.employeeActivity.create({
      data: {
        employeeId,
        userId,
        action: 'document_added',
        changes: { fileName: { old: null, new: file.name } },
      },
    })

    return NextResponse.json({ document })
  })
}
```

```typescript
// app/api/employees/documents/[id]/route.ts
import { NextResponse } from 'next/server'
import { withWorkspace } from '@/lib/auth/with-workspace'
import { checkPermission, Permission } from '@/lib/permissions/roles'
import { WorkspaceRole } from '@prisma/client'
import { prisma } from '@/lib/prisma'
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
)

export async function DELETE(
  request: Request,
  { params }: { params: { id: string } }
) {
  return withWorkspace(async ({ workspaceId, userId, role }) => {
    checkPermission(role as WorkspaceRole, Permission.MANAGE_EMPLOYEES)

    const document = await prisma.employeeDocument.findFirst({
      where: {
        id: params.id,
        employee: { workspaceId },
      },
      include: {
        employee: true,
      },
    })

    if (!document) {
      return NextResponse.json({ error: 'Document not found' }, { status: 404 })
    }

    // Extract file path from URL
    const urlParts = document.fileUrl.split('/employee-documents/')
    const filePath = urlParts[1]

    // Delete from Supabase Storage
    if (filePath) {
      await supabase.storage.from('employee-documents').remove([filePath])
    }

    // Delete database record
    await prisma.employeeDocument.delete({
      where: { id: params.id },
    })

    // Log activity
    await prisma.employeeActivity.create({
      data: {
        employeeId: document.employeeId,
        userId,
        action: 'document_deleted',
        changes: { fileName: { old: document.fileName, new: null } },
      },
    })

    return NextResponse.json({ success: true })
  })
}
```
