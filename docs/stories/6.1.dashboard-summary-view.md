# Story 6.1: Build Dashboard Summary View

## Status
Draft

## Story

**As a** user,
**I want** to see a dashboard when I log in,
**so that** I get an overview of my compliance status and priorities.

## Acceptance Criteria

1. Dashboard page created at `/dashboard` (default landing after login)
2. **Compliance Progress Ring:** Circular progress chart showing % of laws "Compliant" vs total
3. **AI Insights Section:**
   - Recent law changes (last 7 days) affecting workspace
   - New laws recommended for industry
   - AI-generated priority suggestions: "3 laws need urgent attention"
4. **Quick Actions:** Buttons for "Ask AI", "Add Law", "Invite Team", "Add Employee"
5. **Recent Activity Feed:**
   - Last 10 actions: "Anna reviewed Law X", "Law Y changed yesterday"
   - Timestamp + user avatar
6. **Law List Preview:** Top 5 prioritized laws with status badges
7. Mobile-responsive layout (stacked sections on mobile)
8. Dashboard loads in <2 seconds

**Dashboard Stats Helper:**
```typescript
// lib/dashboard/get-stats.ts
import { prisma } from '@/lib/prisma'
import { withWorkspace } from '@/lib/auth/with-workspace'

export async function getDashboardStats() {
  return withWorkspace(async ({ workspaceId }) => {
    // Get law status counts
    const lawsByStatus = await prisma.workspaceLaw.groupBy({
      by: ['status'],
      where: { workspaceId },
      _count: true
    })

    const statusMap = lawsByStatus.reduce((acc, item) => {
      acc[item.status] = item._count
      return acc
    }, {} as Record<string, number>)

    // Get total law count
    const totalLaws = await prisma.workspaceLaw.count({
      where: { workspaceId }
    })

    // Get recent law changes (last 7 days)
    const sevenDaysAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000)
    const recentChanges = await prisma.lawChange.count({
      where: {
        publishedDate: { gte: sevenDaysAgo },
        law: {
          workspaceLaws: {
            some: { workspaceId }
          }
        }
      }
    })

    // Get high priority law count
    const highPriorityCount = await prisma.workspaceLaw.count({
      where: {
        workspaceId,
        priority: 'high',
        status: { not: 'compliant' }
      }
    })

    return {
      lawsByStatus: statusMap,
      totalLaws,
      recentChanges,
      highPriorityCount
    }
  })
}
```

**AI Insights Helper:**
```typescript
// lib/dashboard/get-ai-insights.ts
import { prisma } from '@/lib/prisma'
import { withWorkspace } from '@/lib/auth/with-workspace'

export async function getAIInsights() {
  return withWorkspace(async ({ workspaceId, workspace }) => {
    const sevenDaysAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000)

    // Get recent law changes affecting this workspace
    const recentChanges = await prisma.lawChange.findMany({
      where: {
        publishedDate: { gte: sevenDaysAgo },
        law: {
          workspaceLaws: {
            some: { workspaceId }
          }
        }
      },
      include: {
        law: {
          select: {
            title: true,
            sfsNumber: true
          }
        }
      },
      orderBy: { publishedDate: 'desc' },
      take: 5
    })

    // Get high-priority laws that need attention
    const prioritySuggestions = await prisma.workspaceLaw.findMany({
      where: {
        workspaceId,
        priority: 'high',
        status: { in: ['not_started', 'blocked'] }
      },
      include: {
        law: {
          select: {
            title: true,
            sfsNumber: true
          }
        }
      },
      take: 5
    })

    // Get recommended laws based on industry
    // (Simplified: Find laws in starter pack not yet added)
    const existingLawIds = await prisma.workspaceLaw.findMany({
      where: { workspaceId },
      select: { lawId: true }
    }).then((laws) => laws.map((l) => l.lawId))

    const recommendedLaws = await prisma.law.findMany({
      where: {
        id: { notIn: existingLawIds },
        categories: {
          some: {
            category: {
              name: { in: ['Arbetsmiljö', 'Arbetsrätt', 'Grundläggande'] }
            }
          }
        }
      },
      take: 3
    })

    return {
      recentChanges: recentChanges.map((change) => ({
        id: change.id,
        lawTitle: change.law.title,
        lawSfsNumber: change.law.sfsNumber,
        summary: change.summary,
        publishedDate: change.publishedDate
      })),
      prioritySuggestions: prioritySuggestions.map((wl) => ({
        id: wl.id,
        lawTitle: wl.law.title,
        lawSfsNumber: wl.law.sfsNumber,
        status: wl.status,
        priority: wl.priority
      })),
      recommendedLaws: recommendedLaws.map((law) => ({
        id: law.id,
        title: law.title,
        sfsNumber: law.sfsNumber
      }))
    }
  })
}
```

```typescript
import { prisma } from '@/lib/prisma'
import { withWorkspace } from '@/lib/auth/with-workspace'

export async function getRecentActivity() {
  return withWorkspace(async ({ workspaceId }) => {
    const activities = await prisma.workspaceActivityLog.findMany({
      where: { workspaceId },
      include: {
        user: {
          select: {
            email: true,
            name: true
          }
        }
      },
      orderBy: { timestamp: 'desc' },
      take: 10
    })

    return activities.map((activity) => ({
      id: activity.id,
      action: activity.action,
      description: getActivityDescription(activity),
      userEmail: activity.user.email,
      userName: activity.user.name,
      timestamp: activity.timestamp
    }))
  })
}

function getActivityDescription(activity: any): string {
  const metadata = activity.metadata || {}

  switch (activity.action) {
    case 'law_reviewed':
      return `reviewed "${metadata.lawTitle || 'a law'}"`
    case 'law_added':
      return `added "${metadata.lawTitle || 'a law'}" to workspace`
    case 'law_removed':
      return `removed "${metadata.lawTitle || 'a law'}" from workspace`
    case 'law_status_changed':
      return `moved "${metadata.lawTitle || 'a law'}" to ${metadata.newStatus || 'new status'}`
    case 'employee_added':
      return `added employee "${metadata.employeeName || 'an employee'}"`
    case 'employee_updated':
      return `updated employee "${metadata.employeeName || 'an employee'}"`
    case 'employee_deleted':
      return `deleted employee "${metadata.employeeName || 'an employee'}"`
    case 'member_invited':
      return `invited ${metadata.inviteeEmail || 'a team member'} as ${metadata.role || 'member'}`
    case 'member_joined':
      return 'joined the workspace'
    case 'settings_updated':
      return 'updated workspace settings'
    default:
      return activity.action.replace(/_/g, ' ')
  }
}
```

```typescript
// lib/dashboard/get-top-priority-laws.ts
import { prisma } from '@/lib/prisma'
import { withWorkspace } from '@/lib/auth/with-workspace'

export async function getTopPriorityLaws() {
  return withWorkspace(async ({ workspaceId }) => {
    const laws = await prisma.workspaceLaw.findMany({
      where: {
        workspaceId,
        status: { not: 'compliant' }
      },
      include: {
        law: {
          include: {
            categories: {
              include: {
                category: true
              },
              take: 1
            }
          }
        }
      },
      orderBy: [
        { priority: 'desc' }, // high > medium > low
        { addedAt: 'desc' }
      ],
      take: 5
    })

    return laws.map((wl) => ({
      id: wl.id,
      lawId: wl.lawId,
      lawTitle: wl.law.title,
      lawSfsNumber: wl.law.sfsNumber,
      category: wl.law.categories[0]?.category.name || 'Övrigt',
      status: wl.status,
      priority: wl.priority
    }))
  })
}
```
